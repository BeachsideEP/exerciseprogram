🔒 Step 3 — Enable Row-Level Security (Important!)
Run this SQL once in your Supabase SQL Editor to lock down access — practitioners see everything, clients only see their own record: 
-- Run once in Supabase SQL Editor

-- 1. Create tables
create table if not exists clients  (id text primary key, data jsonb, updated_at timestamptz);
create table if not exists templates (id text primary key, data jsonb, created_at timestamptz);
create table if not exists library   (id text primary key, data jsonb, updated_at timestamptz);

-- 2. Enable Row Level Security
alter table clients   enable row level security;
alter table templates enable row level security;
alter table library   enable row level security;

-- 3. Practitioners: full access to everything
create policy "practitioners_clients"   on clients   for all using (auth.email() like '%@beachsideep.com.au') with check (auth.email() like '%@beachsideep.com.au');
create policy "practitioners_templates" on templates for all using (auth.email() like '%@beachsideep.com.au') with check (auth.email() like '%@beachsideep.com.au');
create policy "practitioners_library"   on library   for all using (auth.email() like '%@beachsideep.com.au') with check (auth.email() like '%@beachsideep.com.au');

-- 4. Clients: read their own record only
create policy "clients_own_record" on clients for select using (data->>'email' = auth.email());

-- 5. Clients: read library (so they see custom exercise photos)
create policy "clients_read_library" on library for select using (auth.role() = 'authenticated');
⚠️ Without this, any client who downloads the file can read all patient data
